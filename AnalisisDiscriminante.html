#Seguidament, calcula un valor reescalat de la dist`ancia D2 de Mahalanobis al quadrat entre els dos centroides(mostrals) i el valor de l’estad´ıstic T^2
D2 <- matrix(m1-m2,nrow=1) %*% solve(Spool) %*% matrix(m1-m2,ncol=1)
T2<- D2*(n1*n2)/(n1+n2)
